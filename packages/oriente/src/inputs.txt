
interface InputProps<T = string> {
  value: T
  onChange?: (value: T) => void
  isDisabled?: boolean
  onFocus?: () => void
  onBlur?: () => void
}

interface TextInputProps extends InputProps {
  isMultiline?: boolean
  lines?: number
  maxLines?: number
  // input element props
}

interface ToggleInputProps extends InputProps<boolean> {}

interface SliderInputProps extends InputProps<number> {
  min?: number
  max?: number
  step?: number
  orientation?: 'horiz' | 'vert'
  // range
  // marks
  // label
}
